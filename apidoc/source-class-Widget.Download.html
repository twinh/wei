33:      */
34:     protected $disposition = 'attachment';

51:     /**
52:      * Send file download response
53:      *
54:      * @param string $file The path of file
55:      * @param array $options The widget options
56:      * @return Download
57:      * @throws \RuntimeException When file not found
58:      */
59:     public function send($file = null, $options = array())
60:     {
61:         $options && $this->setOption($options);
62: 
63:         if (!is_file($file)) {
64:             throw new \RuntimeException('File not found', 404);
65:         }
66: 
67:         $name = $this->filename ?: basename($file);
68: 
69:         // For IE
70:         if (preg_match('/MSIE ([\w.]+)/', $this->request->getServer('HTTP_USER_AGENT'))) {
71:             $filename = '=' . rawurlencode($name);
72:         } else {
73:             $filename = "*=UTF-8''" . rawurlencode($name);
74:         }
75: 
76:         $this->header->set(array(
77:             'Content-Description'       => 'File Transfer',
78:             'Content-Type'              => $this->type,
79:             'Content-Disposition'       => $this->disposition . ';filename' . $filename,
80:             'Content-Transfer-Encoding' => 'binary',
81:             'Expires'                   => '0',
82:             'Cache-Control'             => 'must-revalidate',
83:             'Pragma'                    => 'public',
84:             'Content-Length'            => filesize($file),
85:         ));
86: 
87:         // Send headers
88:         parent::send();
89: 
90:         // Send file content
