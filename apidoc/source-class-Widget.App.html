 84:     /**
 85:      * Load and execute the controller action
 86:      *
 87:      * @param  string    $controller The name of controller
 88:      * @param  string    $action     The name of action
 89:      * @return App
 90:      * @throws \RuntimeException When controller or action not found
 91:      */
 92:     public function dispatch($controller, $action = 'index')
 93:     {
 94:         $object = $this->getControllerInstance($controller, $action);
 95:         if ($object) {
 96:             // Check if action is exists and public
 97:             if (method_exists($object, $action)) {
 98:                 $ref = new \ReflectionMethod($object, $action);
 99:                 if ($ref->isPublic()) {
100:                     try {
101:                         $response = $object->$action();
102:                         $this->handleResponse($response);
103:                     } catch (\RuntimeException $e) {
104:                         if ($e->getCode() === self::FORWARD_CODE) {
105:                             $this->logger->debug(sprintf('Caught exception "%s" with message "%s" called in %s on line %s', get_class($e), $e->getMessage(), $e->getFile(), $e->getLine()));
106:                         } else {
107:                             throw $e;
108:                         }
109:                     }
110:                     return $this;
111:                 } else {
112:                     $notFound = 'action';
113:                 }
114:             } else {
115:                 $notFound = 'action';
116:             }
117:         } else {
118:             $notFound = 'controller';
119:         }
120: 
121:         // Prepare exception message
122:         $message = 'The page you requested was not found';
123:         if ($this->widget->inDebug()) {
124:             $message .= ' - ';
125:             switch ($notFound) {
126:                 case 'controller':
127:                     $message .= sprintf('controller "%s" (class "%s") not found', $controller, $this->getControllerClass($controller));
128:                     break;
129: 
130:                 case 'action':
131:                     $message .= sprintf('action method "%s" not found in controller "%s" (class "%s")', $action, $controller, get_class($object));
132:                     break;
133:             }
134:         }
135:         // You can use `$widget->error->notFound(function(){});` to custom the 404 page
136:         throw new \RuntimeException($message, 404);
137:     }

139:     /**
140:      * Handle the response variable returned by controller action
141:      *
142:      * @param  mixed                     $response
143:      * @return Response|boolean
144:      * @throws \InvalidArgumentException
145:      */
146:     public function handleResponse($response)
147:     {
148:         switch (true) {
149:             // Render default template and using $result as template variables
150:             case is_array($response) :
151:                 $response = $this->view->render($this->getDefaultTemplate(), $response);
152:                 // No break here, $response is use for next case detect
153: 
154:             // Response directly
155:             case is_string($response) :
156:             case is_null($response) :
157:                 return $this->response($response);
158: 
159:             // Response if not sent
160:             case $response instanceof Response :
161:                 return !$response->isSent() && $response->send();
162: 
163:             default :
164:                 throw new \InvalidArgumentException(sprintf(
165:                     'Expected argument of type array, printable variable or \Widget\Response, "%s" given',
166:                     is_object($response) ? get_class($response) : gettype($response)
167:                 ));
168:         }
169:     }

245:     public function getControllerInstance($controller, $action)
246:     {
247:         $class = $this->getControllerClass($controller);

249:         if (isset($this->controllers[$class])) {

253:         if (!class_exists($class)) {
254:             return false;
